def social_media_icon_color_hover
  if community.custom_color1.present?
    (0..2).map{ |x| community.custom_color1.slice(x*2, 2).to_i(16) }.join(',')
  else
    (theme_dark? ? '217,217,217' : '74,144,226')
  end
end
