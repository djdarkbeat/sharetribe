def build(data)
  with_result(
    specs: @specs,
    data: data,
    on_success: ->(result) {
      result[:value]
    },
    on_failure: ->(result) {
      loc = caller_locations(4, 1).first
      raise(ArgumentError, "Error(s) in #{loc}: #{error_msg(result)}")
    })
end

def validate(data)
  with_result(
    specs: @specs,
    data: data,
    on_success: ->(result) {
      Result::Success.new(result[:value])
    },
    on_failure: ->(result) {
      Result::Error.new(error_msg(result), result[:errors])
    })
end
