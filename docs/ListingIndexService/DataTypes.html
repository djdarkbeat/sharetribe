def parse_fields(fields)
  (fields || []).map { |f|
    case f[:type]
    when :numeric_range
      NumericRange.call(f)
    when :selection_group
      SelectionGroup.call(f)
    else
      ArgumentError.new("Unknown field type: #{f[:type]}")
    end
  }
end
