def call(type, id, _)
  unless @_data.key?(id)
    raise LinkResolvingError.new("Unknown marketplace data value '#{id}'.")
  end

  value =
    case id
    when "search_type"
      search_type
    else
      value = @_data[id]
    end

  { "id" => id, "type" => type, "value" => value }
end
