def call(type, id, normalized_data)
  asset = normalized_data[type].find { |item| item["id"] == id }

  if asset.nil?
    raise LinkResolvingError.new("Unable to find an asset with id '#{id}'.")
  elsif asset['absolute_path']
    asset
  else
    append_asset_path(asset)
  end
end
