def shape_direction_map(shapes, processes)
  ArrayUtils.inner_join(shapes, processes) { |shape, process|
    shape[:transaction_process_id] == process[:id]
  }.map { |(shape, process)|
    [shape[:id], process_to_direction(process)]
  }.to_h
end
