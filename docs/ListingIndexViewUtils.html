def to_struct(result:, includes:, per_page:, page:)
  listings = result[:listings].map { |l|
    author =
      if includes.include?(:author)
        Author.new(
          l[:author][:id],
          l[:author][:username],
          l[:author][:first_name],
          l[:author][:last_name],
          l[:author][:display_name],
          ListingImage.new(
            l[:author][:avatar][:thumb]
          ),
          l[:author][:is_deleted],
          l[:author][:num_of_reviews]
        )
      end

    listing_images =
      if includes.include?(:listing_images)
        l[:listing_images].map { |li|
          ListingImage.new(li[:thumb], li[:small_3x2], li[:medium])
        }
      else
        []
      end

    ListingItem.new(
      l[:id],
      l[:url],
      l[:title],
      l[:category_id],
      l[:latitude],
      l[:longitude],
      l[:distance],
      l[:distance_unit],
      author,
      l[:description],
      listing_images,
      l[:price],
      l[:unit_tr_key],
      l[:unit_type],
      l[:quantity],
      l[:shape_name_tr_key],
      l[:listing_shape_id],
      l[:icon_name]
    )
  }

  paginated = WillPaginate::Collection.create(page, per_page, result[:count]) do |pager|
    pager.replace(listings)
  end
end
