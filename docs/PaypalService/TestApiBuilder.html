  # Empty queue implicitly means :ok
  @next_responses = []
end

def call(req)
  res_type = @next_responses.shift
  if (res_type.is_a? String)
    TestApi.new(req[:receiver_username], true, res_type)
  else
    TestApi.new(req[:receiver_username])
  end
end
